RPi4Toolbox class structure

class Subject
    Subject.id
    Subject.sex
    Subject.age
    Subject.features
    Subject.group
    Subject.progress
    Subject.next_session

class Session
    Session.name = Subject.next_session
    Session.condition
    Session.num_trials
    Session.start (runs trial.start in a loop)
    def end()
        save metadata

class Trial 
    Trial.count
    Trial.reps
    def start()
        led.signal
        key.callback
        choice.eval
        feeder.reward
    def end()
        save metadata
